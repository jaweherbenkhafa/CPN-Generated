\EVT{AddLandingAircraft}{false}{ordinary}{}{}{
	\ANY{
		\Param{l}{true}{}
	}
	\GUARDS{true}{
		\Guard{grd1}{false}{$l \in{} AIRCRAFTS \setminus{} LandingAircraft $}{true}{}
		\Guard{grd2}{false}{$l \notin{} TakeOffAircraft$}{true}{}
	}
	\ACTIONS{true}{
		\Action{act1}{$LandingAircraft\bcmeq{}  LandingAircraft \bunion{} \{l\}$}{true}{}
	}
}
\EVT{AddLandingSequence}{false}{ordinary}{}{}{
	\ANY{
		\Param{l}{true}{}
		\Param{arrm}{true}{}
		\Param{arrh}{true}{}
	}
	\GUARDS{true}{
		\Guard{grd1}{false}{$l\in{} LandingAircraft $}{true}{}
		\Guard{grd2}{false}{$arrh \in{} Hours$}{true}{}
		\Guard{grd3}{false}{$arrm \in{} Minute$}{true}{}
		\Guard{grd4}{false}{$l \notin{} dom(ArrivalM)$}{true}{}
		\Guard{grd5}{false}{$curTimeMin(arrm, arrh) > curTimeMin(CurtM, CurtH)$}{true}{}
		\Guard{grd6}{false}{$curTimeMin(arrm,arrh) \leq{} curTimeMin(CurtM,CurtH) + zoomlevel$}{true}{}
		\Guard{grd7}{false}{$(arrh \mapsto{} arrm) \in{} (Hours \cprod{} Minute)$}{true}{}
		\Guard{grd8}{false}{$l \notin{} dom(LandingSequence)$}{true}{}
	}
	\ACTIONS{true}{
		\Action{act1}{$ArrivalM\bcmeq{} ArrivalM \bunion{} \{l \mapsto{} arrm\}$}{true}{}
		\Action{act2}{$ArrivalH\bcmeq{} ArrivalH \bunion{} \{l \mapsto{} arrh\}$}{true}{}
		\Action{act3}{$LandingSequence \bcmeq{} LandingSequence \bunion{} \{ l \mapsto{} (arrh \mapsto{} arrm) \}$}{true}{}
	}
}
\EVT{DeleteLandingAircraft}{false}{ordinary}{}{}{
	\ANY{
		\Param{l}{true}{}
		\Param{arrm}{true}{}
		\Param{arrh}{true}{}
	}
	\GUARDS{true}{
		\Guard{grd1}{false}{$l \in{} dom(LandingSequence)$}{true}{}
		\Guard{grd2}{false}{$arrh  \in{} \intg{}$}{true}{}
		\Guard{grd3}{false}{$arrm  \in{} \intg{}$}{true}{}
		\Guard{grd4}{false}{$arrh \mapsto{} arrm = LandingSequence(l)$}{true}{}
		\Guard{grd5}{false}{$curTimeMin(arrm, arrh) < curTimeMin(CurtM, CurtH)$}{true}{}
	}
	\ACTIONS{true}{
		\Action{act1}{$LandingAircraft\bcmeq{} LandingAircraft \setminus{} \{l\}$}{true}{}
		\Action{act2}{$ArrivalM \bcmeq{} \{l\} \domsub{} ArrivalM$}{true}{}
		\Action{act3}{$ArrivalH \bcmeq{} \{l\} \domsub{} ArrivalH$}{true}{}
		\Action{act4}{$LandingSequence \bcmeq{} \{l\} \domsub{} LandingSequence$}{true}{}
	}
}

\EVT{AddTakeOffSequence}{false}{ordinary}{}{}{
	\ANY{
		\Param{a}{true}{}
		\Param{decom}{true}{}
		\Param{decoh}{true}{}
	}
	\GUARDS{true}{
		\Guard{grd1}{false}{$a \in{} TakeOffAircraft$}{true}{}
		\Guard{grd2}{false}{$decoh  \in{} Hours \land{} decoh \in{} \intg{}$}{true}{}
		\Guard{grd3}{false}{$decom  \in{} Minute \land{} decom \in{} \intg{}$}{true}{}
		\Guard{grd4}{false}{$a \notin{} dom(DepM)$}{true}{}
		\Guard{grd5}{false}{$curTimeMin(decom, decoh) \geq{} curTimeMin(CurtM, CurtH)$}{true}{}
		\Guard{grd6}{false}{$\forall{}l1\qdot{}l1 \in{} dom(DepM) \limp{} (curTimeMin(decom,decoh)-curTimeMin(DepM(l1),DepH(l1))) \geq{} sep$}{true}{}
		\Guard{grd7}{false}{$curTimeMin(decom,decoh) \leq{} curTimeMin(CurtM, CurtH) + zoomlevel$}{true}{}
		\Guard{grd8}{false}{$a \notin{}  dom(TakeOffSequence)$}{true}{}
	}
	\ACTIONS{true}{
		\Action{act1}{$DepM\bcmeq{} DepM \bunion{} \{a \mapsto{} decom\}$}{true}{}
		\Action{act2}{$DepH\bcmeq{} DepH \bunion{} \{a \mapsto{} decoh\}$}{true}{}
		\Action{act3}{$TakeOffSequence \bcmeq{} TakeOffSequence \bunion{} \{a \mapsto{} (decoh \mapsto{} decom)\}$}{true}{}
	}
}

	}
